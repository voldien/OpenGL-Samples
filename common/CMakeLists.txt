FILE(GLOB SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_LIBRARY(Common STATIC ${SOURCE_FILES} ${HEADER_FILES} )
TARGET_LINK_LIBRARIES(Common ${OPENGL_LIBRARIES} ${GLEW_LIBRARIES} ${SDL2_LINK_LIBRARIES})
# OpenGL dependies headers.
TARGET_INCLUDE_DIRECTORIES(Common PRIVATE ${OPENGL_INCLUDE_DIR} ${GLEW_INCLUDE_DIRS} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} )
