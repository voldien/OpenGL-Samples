FILE(GLOB SOURCE_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
FILE(GLOB HEADER_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

ADD_EXECUTABLE(SDLGLWindow  ${SOURCE_FILES} ${HEADER_FILES} )
TARGET_LINK_LIBRARIES(SDLGLWindow ${OPENGL_LIBRARIES} ${GLEW_LIBRARIES} ${SDL2_LINK_LIBRARIES} glCommon)

# OpenGL dependies headers.
TARGET_INCLUDE_DIRECTORIES(SDLGLWindow PRIVATE ${OPENGL_INCLUDE_DIR} ${GLEW_INCLUDE_DIRS})
ADD_DEPENDENCIES(SDLGLWindow glCommon)
